📋 Guide de configuration du Google Sheet
1. Créer le Google Sheet
Créez un Google Sheet avec 3 onglets :
• Voyages — colonnes : Nom | Lieu | Latitude | Longitude | DateDebut | DateFin | Description
• Meetups — colonnes : ID | Proposeur | Lieu | Latitude | Longitude | Date | Description | Participants
• Config — cellule A1 = mot de passe admin (ex: vacamap2025)

2. Rendre le Sheet accessible
• Fichier → Partager → "Tous ceux qui ont le lien" → Lecteur (suffit pour la lecture)
• Pour la lecture ET l'écriture → Éditeur

3. Clé API Google (pour la lecture)
• console.cloud.google.com → Créer un projet
• APIs → Bibliothèque → activer Google Sheets API
• Identifiants → Créer une clé API → Copier

4. Web App Apps Script (pour l'écriture)
• Dans le Sheet → Extensions → Apps Script → Coller le code ci-dessous
• Déployer → Nouvelle version → Type: "Application Web" → Accès: "Tout le monde"
• Copier l'URL de déploiement

📜 Code Apps Script à copier 
function doPost(e) {
  var lock = LockService.getScriptLock();
  lock.tryLock(10000);
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var action = data.action;

    if (action === "addTrip") {
      var sheet = ss.getSheetByName("Voyages");
      sheet.appendRow([data.nom, data.lieu, data.lat, data.lng, data.dateDebut, data.dateFin, data.description]);
      return ok({status: "ok"});
    }
    if (action === "deleteTrip") {
      var sheet = ss.getSheetByName("Voyages");
      var rows = sheet.getDataRange().getValues();
      for (var i = rows.length - 1; i >= 1; i--) {
        if (rows[i][0] === data.nom && rows[i][1] === data.lieu && rows[i][4] === data.dateDebut) sheet.deleteRow(i + 1);
      }
      return ok({status: "ok"});
    }
    if (action === "deleteProfile") {
      var s1 = ss.getSheetByName("Voyages"); var r1 = s1.getDataRange().getValues();
      for (var i = r1.length - 1; i >= 1; i--) { if (r1[i][0] === data.nom) s1.deleteRow(i + 1); }
      var s2 = ss.getSheetByName("Meetups"); var r2 = s2.getDataRange().getValues();
      for (var i = r2.length - 1; i >= 1; i--) { if (r2[i][1] === data.nom) s2.deleteRow(i + 1); }
      return ok({status: "ok"});
    }
    if (action === "addMeetup") {
      var sheet = ss.getSheetByName("Meetups");
      sheet.appendRow([data.id, data.proposeur, data.lieu, data.lat, data.lng, data.date, data.description, data.participants]);
      return ok({status: "ok"});
    }
    if (action === "joinMeetup") {
      var sheet = ss.getSheetByName("Meetups"); var rows = sheet.getDataRange().getValues();
      for (var i = 1; i < rows.length; i++) {
        if (rows[i][0] === data.id) {
          var parts = rows[i][7] ? rows[i][7].toString().split(",").map(function(s){return s.trim();}).filter(Boolean) : [];
          if (parts.indexOf(data.nom) === -1) parts.push(data.nom);
          sheet.getRange(i + 1, 8).setValue(parts.join(", "));
        }
      }
      return ok({status: "ok"});
    }
    if (action === "deleteMeetup") {
      var sheet = ss.getSheetByName("Meetups"); var rows = sheet.getDataRange().getValues();
      for (var i = rows.length - 1; i >= 1; i--) { if (rows[i][0] === data.id) sheet.deleteRow(i + 1); }
      return ok({status: "ok"});
    }
    return ok({status: "error", message: "Unknown action"});
  } catch(err) {
    return ok({status: "error", message: err.toString()});
  } finally { lock.releaseLock(); }
}
function ok(data) {
  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
}
function doGet(e) { return ok({status: "VacaMap API is running"}); }
