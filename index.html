Connexion Google Sheet

Votre identité
Utilisée pour tracer vos trajets et détecter des croisements avec vos amis
📋 Guide de configuration
1. Créer le Google Sheet avec les onglets suivants :
• Voyages — 17 colonnes (créé automatiquement par le script au 1er ajout) :
  Nom | LieuDepart | LatDepart | LngDepart | DateDepart | TransportAller | DateArrivee | Lieu | Lat | Lng | DateDepartRetour | LieuArrivee | LatArrivee | LngArrivee | TransportRetour | Description | TripId
  ⚠️ Si vous avez une ancienne feuille "Voyages" (8 colonnes), supprimez-la — elle sera recréée automatiquement.
• Meetups — ID | Proposeur | Lieu | Latitude | Longitude | Date | Description | Participants
• Profils — créé automatiquement
• Messages — créé automatiquement
• Config — cellule A2 = mot de passe admin (A1 = en-tête)

2. Rendre le Sheet public : Partager → Lecteur → Tout le monde

3. Clé API : console.cloud.google.com → Google Sheets API → Identifiants

4. Apps Script : Extensions → Apps Script → Coller le code → Déployer en Web App (Tout le monde)
📜 Code Apps Script 
function doPost(e) {
  var lock = LockService.getScriptLock(); lock.tryLock(10000);
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var action = data.action;

    if (action === "upsertProfile") {
      var sh = ss.getSheetByName("Profils");
      if (!sh) { sh = ss.insertSheet("Profils"); sh.appendRow(["Nom","VilleDepart","Lat","Lng"]); }
      var rows = sh.getDataRange().getValues();
      var found = false;
      for (var i = 1; i < rows.length; i++) {
        if (rows[i][0] === data.nom) { sh.getRange(i+1,1,1,4).setValues([[data.nom,data.lieu,data.lat,data.lng]]); found=true; break; }
      }
      if (!found) sh.appendRow([data.nom, data.lieu, data.lat, data.lng]);
      return ok({status:"ok"});
    }
    if (action === "addTrip") {
      var sh = ss.getSheetByName("Voyages");
      sh.appendRow([data.nom,data.lieu,data.lat,data.lng,data.dateDebut,data.dateFin,data.description,data.transport||"voiture"]);
      return ok({status:"ok"});
    }
    if (action === "deleteTrip") {
      var sh = ss.getSheetByName("Voyages"), rows = sh.getDataRange().getValues();
      for (var i=rows.length-1;i>=1;i--) if(rows[i][0]===data.nom&&rows[i][1]===data.lieu&&rows[i][4]===data.dateDebut) sh.deleteRow(i+1);
      return ok({status:"ok"});
    }
    if (action === "deleteProfile") {
      var s1=ss.getSheetByName("Voyages"),r1=s1.getDataRange().getValues();
      for(var i=r1.length-1;i>=1;i--) if(r1[i][0]===data.nom) s1.deleteRow(i+1);
      var s2=ss.getSheetByName("Meetups"),r2=s2.getDataRange().getValues();
      for(var i=r2.length-1;i>=1;i--) if(r2[i][1]===data.nom) s2.deleteRow(i+1);
      var sp=ss.getSheetByName("Profils"); if(sp){var rp=sp.getDataRange().getValues();for(var i=rp.length-1;i>=1;i--)if(rp[i][0]===data.nom)sp.deleteRow(i+1);}
      return ok({status:"ok"});
    }
    if (action === "addMessage") {
      var sh = ss.getSheetByName("Messages");
      if (!sh) { sh = ss.insertSheet("Messages"); sh.appendRow(["Auteur","Message","Timestamp"]); }
      sh.appendRow([data.auteur, data.message, data.timestamp]);
      return ok({status:"ok"});
    }
    if (action === "addMeetup") {
      var sh = ss.getSheetByName("Meetups");
      sh.appendRow([data.id,data.proposeur,data.lieu,data.lat,data.lng,data.date,data.description,data.participants]);
      return ok({status:"ok"});
    }
    if (action === "joinMeetup") {
      var sh=ss.getSheetByName("Meetups"),rows=sh.getDataRange().getValues();
      for(var i=1;i=1;i--) if(rows[i][0]===data.id) sh.deleteRow(i+1);
      return ok({status:"ok"});
    }
    return ok({status:"error",message:"Unknown action"});
  } catch(err) { return ok({status:"error",message:err.toString()}); }
  finally { lock.releaseLock(); }
}
function ok(d){return ContentService.createTextOutput(JSON.stringify(d)).setMimeType(ContentService.MimeType.JSON);}
function doGet(e){return ok({status:"VacaMap API running"});}
