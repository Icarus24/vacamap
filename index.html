// ══════════════════════════════════════════════════════ // VacaMap — Google Apps Script // À coller dans : Google Sheet → Extensions → Apps Script // Puis : Déployer → Nouveau déploiement → Application Web // → Exécuter en tant que : Moi // → Accès : Tout le monde // // ⚠️ IMPORTANT — Nouveau schéma de la feuille "Voyages" (17 colonnes) : // A=Nom, B=LieuDepart, C=LatDepart, D=LngDepart, // E=DateDepart, F=TransportAller, G=DateArrivee, // H=Lieu (destination), I=Lat, J=Lng, // K=DateDepartRetour, L=LieuArrivee, M=LatArrivee, N=LngArrivee, // O=TransportRetour, P=Description, Q=TripId // // → Si vous aviez une ancienne feuille Voyages (8 colonnes), // supprimez-la et laissez le script la recréer automatiquement. // ══════════════════════════════════════════════════════ function doPost(e) { var lock = LockService.getScriptLock(); lock.tryLock(10000); try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var action = data.action; // ── Profils (ville de départ) ────────────────────── if (action === "upsertProfile") { var sh = ss.getSheetByName("Profils"); if (!sh) { sh = ss.insertSheet("Profils"); sh.appendRow(["Nom", "VilleDepart", "Lat", "Lng"]); } var rows = sh.getDataRange().getValues(); var found = false; for (var i = 1; i < rows.length; i++) { if (rows[i][0] === data.nom) { sh.getRange(i + 1, 1, 1, 4).setValues([[data.nom, data.lieu, data.lat, data.lng]]); found = true; break; } } if (!found) sh.appendRow([data.nom, data.lieu, data.lat, data.lng]); return ok({ status: "ok" }); } // ── Voyages ──────────────────────────────────────── if (action === "addTrip") { var sh = ss.getSheetByName("Voyages"); if (!sh) { sh = ss.insertSheet("Voyages"); sh.appendRow(["Nom","LieuDepart","LatDepart","LngDepart","DateDepart","TransportAller", "DateArrivee","Lieu","Lat","Lng","DateDepartRetour","LieuArrivee","LatArrivee", "LngArrivee","TransportRetour","Description","TripId"]); } var tripId = data.tripId || ("T" + new Date().getTime().toString(36)); sh.appendRow([ data.nom, data.lieuDepart || "", data.latDepart || 0, data.lngDepart || 0, data.dateDepart || "", data.transportAller || "voiture", data.dateArrivee || "", data.lieu || "", data.lat || 0, data.lng || 0, data.dateDepartRetour || "", data.lieuArrivee || "", data.latArrivee || 0, data.lngArrivee || 0, data.transportRetour || "voiture", data.description || "", tripId ]); return ok({ status: "ok", tripId: tripId }); } if (action === "deleteTrip") { var sh = ss.getSheetByName("Voyages"); if (!sh) return ok({ status: "ok" }); var rows = sh.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { if (rows[i][0].toString() !== data.nom) continue; // Priorité : correspondance par TripId (colonne Q = index 16) if (data.tripId && rows[i][16] && rows[i][16].toString() === data.tripId.toString()) { sh.deleteRow(i + 1); break; } // Fallback : correspondance par lieu destination (pour anciennes données) if (!data.tripId && data.lieu && rows[i][7].toString() === data.lieu.toString()) { sh.deleteRow(i + 1); break; } } return ok({ status: "ok" }); } if (action === "deleteProfile") { // Supprimer les voyages var s1 = ss.getSheetByName("Voyages"); if (s1) { var r1 = s1.getDataRange().getValues(); for (var i = r1.length - 1; i >= 1; i--) { if (r1[i][0] === data.nom) s1.deleteRow(i + 1); } } // Supprimer les meetups var s2 = ss.getSheetByName("Meetups"); if (s2) { var r2 = s2.getDataRange().getValues(); for (var i = r2.length - 1; i >= 1; i--) { if (r2[i][1] === data.nom) s2.deleteRow(i + 1); } } // Supprimer le profil var sp = ss.getSheetByName("Profils"); if (sp) { var rp = sp.getDataRange().getValues(); for (var i = rp.length - 1; i >= 1; i--) { if (rp[i][0] === data.nom) sp.deleteRow(i + 1); } } return ok({ status: "ok" }); } // ── Messages (chat) ──────────────────────────────── if (action === "addMessage") { var sh = ss.getSheetByName("Messages"); if (!sh) { sh = ss.insertSheet("Messages"); sh.appendRow(["Auteur", "Message", "Timestamp"]); } sh.appendRow([data.auteur, data.message, data.timestamp]); return ok({ status: "ok" }); } // ── Meetups ──────────────────────────────────────── if (action === "addMeetup") { var sh = ss.getSheetByName("Meetups"); sh.appendRow([ data.id, data.proposeur, data.lieu, data.lat, data.lng, data.date, data.description, data.participants ]); return ok({ status: "ok" }); } if (action === "joinMeetup") { var sh = ss.getSheetByName("Meetups"); var rows = sh.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (rows[i][0] === data.id) { var current = rows[i][7] ? rows[i][7].toString() : ""; var parts = current.split(",").map(function(s) { return s.trim(); }).filter(Boolean); if (parts.indexOf(data.nom) === -1) parts.push(data.nom); sh.getRange(i + 1, 8).setValue(parts.join(", ")); } } return ok({ status: "ok" }); } if (action === "deleteMeetup") { var sh = ss.getSheetByName("Meetups"); var rows = sh.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { if (rows[i][0] === data.id) sh.deleteRow(i + 1); } return ok({ status: "ok" }); } return ok({ status: "error", message: "Unknown action" }); } catch (err) { return ok({ status: "error", message: err.toString() }); } finally { lock.releaseLock(); } } function ok(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ok({ status: "VacaMap API running" }); }
