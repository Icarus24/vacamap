📋 Guide de configuration (admin)
1. Créer un projet Supabase
→ supabase.com → New project → choisir un nom et mot de passe DB

2. Créer les tables
→ SQL Editor → New query → coller et exécuter le SQL ci-dessous
📜 SQL à exécuter 
-- Tables VacaMap
create table if not exists profils (id uuid default gen_random_uuid() primary key, nom text not null unique, ville_depart text, lat float8, lng float8, created_at timestamptz default now());
create table if not exists voyages (id uuid default gen_random_uuid() primary key, nom text not null, lieu_depart text, lat_depart float8, lng_depart float8, date_depart date, transport_aller text default 'voiture', date_arrivee date, lieu text, lat float8, lng float8, date_depart_retour date, lieu_arrivee text, lat_arrivee float8, lng_arrivee float8, transport_retour text default 'voiture', description text, trip_id text unique, created_at timestamptz default now());
create table if not exists meetups (id text primary key, proposeur text, lieu text, lat float8, lng float8, date date, description text, participants text, created_at timestamptz default now());
create table if not exists messages (id uuid default gen_random_uuid() primary key, auteur text, message text, timestamp text, created_at timestamptz default now());
create table if not exists config (key text primary key, value text);
insert into config (key, value) values ('admin_password', 'changeme') on conflict (key) do nothing;
-- Accès public (groupe d'amis)
alter table profils enable row level security;
alter table voyages enable row level security;
alter table meetups enable row level security;
alter table messages enable row level security;
alter table config enable row level security;
create policy "all" on profils for all to anon using (true) with check (true);
create policy "all" on voyages for all to anon using (true) with check (true);
create policy "all" on meetups for all to anon using (true) with check (true);
create policy "all" on messages for all to anon using (true) with check (true);
create policy "read" on config for select to anon using (true);
-- Realtime
alter publication supabase_realtime add table voyages;
alter publication supabase_realtime add table meetups;
alter publication supabase_realtime add table messages;

3. Récupérer les clés
→ Settings → API → copier Project URL et anon / public key

4. Générer le code d'invitation
→ Entrer dans l'app en mode avancé → onglet Admin → copier le code → l'envoyer à vos amis

5. Changer le mot de passe admin
→ Dans le SQL Editor : update config set value='votre_mdp' where key='admin_password';
